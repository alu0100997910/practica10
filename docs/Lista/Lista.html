def each
    current=@head
    while current!=nil
        yield current.value
        current=current.next
    end

def push(val)
    if val.instance_of?Array
        val.each do |i|
            aux=Node.new(@tail, i, nil)
            @tail.next=aux
            @tail=aux
        end
    else
        aux=Node.new(@tail, val, nil)
        @tail.next=aux
        @tail=aux
    end
    @tail.value
end

def to_s
    current=@head
    sout="[ "
    while current!=nil
        if current.next==nil
            sout+=current.value.to_s + " "
        else
            sout+=current.value.to_s + ", "
        end
        current=current.next
    end
    sout+="]"
end
