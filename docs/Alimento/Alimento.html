def aibc
    aibc=[]
    @g.each do |value|
        s=[]
        value.each_with_index do |val,j|
            if (j!=0)
                s << (((val-value[0])+(value[j-1]-value[0]))*5)/2
            end
        end
        aibc << s.reduce(:+)
    end
    aibc
end

def indGlu(glucosa)
    indidv=[]
    @g.each_with_index do |value,index|
        indidv << indIdv(index,glucosa)
    end
    (indidv.reduce(:+))/@g.size
end

end
